#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000000;
int dp[201][201];
int go(int n, int k) {
	if (k == 1) return 1;
	int& ret = dp[n][k];
	if (ret != -1) return ret;
	ret = 0;
	for (int i = 0; i <= n; i++) {
		ret = (ret + go(n - i, k - 1)) % MOD;
	}
	return ret;
}

int main() {
	memset(dp, -1, sizeof(dp));
	int n, k;
	scanf("%d %d", &n, &k);
	printf("%d", go(n, k));
	return 0;
}
